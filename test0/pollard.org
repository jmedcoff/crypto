from fractions import gcd
def factor_p(n, f):
    a,b = 2,2
    d = 1
    while d == 1:
        a = f(a)
        b = f(f(a))
        d = gcd(a-b,n)
        print("a: {} b: {} d: {}".format(a,b,d))
        if d > 1 and d < n:
            return d
        if d == n:
            return False
    return False
